name: SonarCloud Analysis

on:
  push:
    branches: [ main, master ]
  pull_request:
    branches: [ main, master ]
  workflow_dispatch: # Allows manual triggering from GitHub Actions tab

jobs:
  sonarcloud:
    name: SonarCloud
    runs-on: ubuntu-latest
    
    steps:
      - uses: actions/checkout@v3
        with:
          fetch-depth: 0  # Shallow clones should be disabled for a better relevancy of analysis
      
      - name: Setup Node.js
        uses: actions/setup-node@v3
        with:
          node-version: '16'
      
      # Cache dependencies to speed up workflow
      - name: Cache node modules
        uses: actions/cache@v3
        with:
          path: |
            **/node_modules
          key: ${{ runner.os }}-modules-${{ hashFiles('**/package-lock.json') }}
      
      # Install dependencies for both client and server
      - name: Install dependencies
        run: |
          cd server && npm install
          cd ../client && npm install
      
      # Run tests and generate coverage reports (adjust as needed for your project)
      - name: Run tests with coverage
        run: |
          cd server && npm test -- --coverage
          cd ../client && npm test -- --coverage
        continue-on-error: true  # Don't fail the workflow if some tests fail
      
      - name: SonarCloud Scan
        uses: SonarSource/sonarcloud-github-action@master
        env:
          GITHUB_TOKEN: ${{ secrets.GITHUB_TOKEN }}  # Needed to get PR information, if any
          SONAR_TOKEN: ${{ secrets.SONAR_TOKEN }}
        with:
          # These parameters match what's in your sonar-project.properties
          args: >
            -Dsonar.projectKey=eli-iie_APDS7311-CustomerPortal
            -Dsonar.organization=eli-iie
